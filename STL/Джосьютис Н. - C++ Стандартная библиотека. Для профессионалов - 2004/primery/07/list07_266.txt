#include <iostream>
#include <list>
#include <algorithm>
using namespace std;

int main()
{
    list<int> coll;

    // Вставка элементов от 1 до 9
    for (int i=1; i<=9; ++i) {
        coll.push_back(i);
    }

    list<int>::iterator pos = coll.begin();

    // Вывод текущего элемента
    cout << *pos << endl;

    // Перемещение вперед на три элемента
    advance (pos, 3);

    // Вывод текущего элемента
    cout << *pos << endl;

    // Перемещение назад на один элемент
    advance (pos, -1);

    // Вывод текущего элемента
    cout << *pos << endl;
}
