// cont/pqueue1.cpp
#include <iostream>
#include <queue>
using namespace std;

int main()
{
    priority_queue<float> q;

    // Вставка трех элементов в приоритетную очередь
    q.push(66.6);
    q.push(22.2);
    q.push(44.4);

    // Вывод и удаление двух элементов
    cout << q.top() << ' ';
    q.pop();
    cout << q.top() << endl;
    q.pop();

    // Вставка еще трех элементов
    q.push(11.1);
    q.push(55.5);
    q.push(33.3);

    // Удаление одного элемента
    q.pop();

    // Извлечение и вывод оставшихся элементов
    while (!q.empty()) {
        cout << q.top() << ' ';
        q.pop();
    }
    cout << endl;
}
