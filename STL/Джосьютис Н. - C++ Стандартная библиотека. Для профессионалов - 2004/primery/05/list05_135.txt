// stl/foreach2.cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

// Простой объект функции для вывода передаваемого аргумента
class PrintInt {
  public:
    void operator() (int elem) const {
        cout << elem << ' ';
    }
};

int main()
{
    vector<int> coll;

    // Вставка элементов со значениями от 1 до 9
    for (int i=1; i<=9; ++i) {
        coll.push_back(i);
    }

    // Вывод всех элементов
    for_each (coll.begin(), coll.end(),    // Интервал
              PrintInt());                 // Операция
    cout << endl;
}
