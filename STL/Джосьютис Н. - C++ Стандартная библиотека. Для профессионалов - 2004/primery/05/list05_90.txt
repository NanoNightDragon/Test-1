// stl/vector1.cpp
#include <iostream>
#include <vector>
using namespace std;

int main()
{
    vector<int> coll;    // Вектор с целыми элементами

    // Присоединение элементов со значениями от 1 до 6
    for (int i=1; i<=6; ++i) {
        coll.push_back(i);
    }

    // Вывод элементов, разделенных пробелами
    for (int i=0; i<coll.size(); ++i) {
        cout << coll[i] << ' ';
    }
    cout << endl;
}
