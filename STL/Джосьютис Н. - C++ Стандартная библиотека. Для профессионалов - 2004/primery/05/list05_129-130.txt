// stl/foreach.cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

// Функция выводит переданный аргумент
void print (int elem)
{
    cout << elem << ' ';
}

int main()
{
    vector<int> coll;

    // Вставка элементов со значениями от 1 до 9
    for (int i=1; i<=9; ++i) {
        coll.push_back(i);
    }

    // Вывод всех элементов
    for_each (coll.begin(), coll.end(),    // Интервал
              print);                      // Операция
    cout << endl;
}
